Key Code Changes Made

Enhanced isBiddingClosed Logic

// New improved logic
const
isBiddingClosed =
  gemstone?.listingStatus === 'sold' ||
  gemstone?.listingStatus === 'expired_no_bids' ||
  (countdownData.isExpired &&
    (bidStats.totalBids > 0 || gemstone?.totalBids > 0));

New State: isWaitingForFirstBid

// New state variable
const
isWaitingForFirstBid =
  !isBiddingClosed &&
  (bidStats.totalBids === 0 &&
    (gemstone?.totalBids === 0 || !gemstone?.totalBids)) &&
  !countdownData.biddingActive;
